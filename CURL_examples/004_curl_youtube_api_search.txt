key = 
url=https://youtube.googleapis.com/youtube/v3/search
q='?part=snippet&maxResults=5&q=Lennon%20Sisters'

##  Construct URL
#   use key=   (for Google)
#   use $key  for oauth2
#   use $api_key for api
URL=${ENDPOINT}?key=${key}
URL=${URL}${PARM}
echo $URL

echo "Running it ..."
# 
curl -v $URL \
  --header 'Accept: application/json' \
  --compressed

exit
